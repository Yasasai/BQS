Task 2 & 3: The Unified Workspace
I created 

UnifiedDashboard.tsx
, a premium component that dynamically adjusts its interface based on the user's role:

GH / PSH: Display global pipeline metrics (Total Value, Stage Counts) and full visibility.
PH / SH: Display assigned opportunities with explicit "Action Required" status flags.
SA / SP: Focused view on technical assessments and assignments.
tsx
/**
 * UnifiedDashboard.tsx
 * Premium consolidated workspace for BQS Enterprise.
 * Fetches data from GET /api/opportunities/ using AuthContext JWT.
 */
import React, { useState, useEffect, useMemo } from 'react';
import axios from 'axios';
import { useAuth, UserRole } from '../context/AuthContext';
import { useNavigate } from 'react-router-dom';
export const UnifiedDashboard: React.FC = () => {
    const { user, token } = useAuth();
    const navigate = useNavigate();
    const [opportunities, setOpportunities] = useState<any[]>([]);
    const [counts, setCounts] = useState<any>({});
    const [metrics, setMetrics] = useState({ total_value: 0, total_count: 0 });
    const [isLoading, setIsLoading] = useState(true);
    const [activeTab, setActiveTab] = useState('ALL');
    const [searchQuery, setSearchQuery] = useState('');
    const fetchDashboardData = async () => {
        if (!user) return;
        setIsLoading(true);
        try {
            const baseUrl = import.meta.env.VITE_API_BASE_URL || '';
            const response = await axios.get(`${baseUrl}/api/opportunities/`, {
                params: {
                    user_id: user.id,
                    role: user.role,
                    tab: activeTab === 'ALL' ? null : activeTab.toLowerCase(),
                    search: searchQuery || null
                }
            });
            setOpportunities(response.data.items || []);
            setCounts(response.data.counts || {});
            setMetrics({
                total_value: response.data.total_value || 0,
                total_count: response.data.total_count || 0
            });
        } catch (err) {
            console.error("Dashboard Fetch Error:", err);
        } finally {
            setIsLoading(false);
        }
    };
    useEffect(() => { fetchDashboardData(); }, [user, activeTab]);
    const isExecutive = user?.role === 'GH' || user?.role === 'PSH';
    const isApprovalRole = user?.role === 'PH' || user?.role === 'SH';
    return (
        <div style={{ padding: '40px', background: '#f8fafc', minHeight: '100vh', fontFamily: "'Outfit', sans-serif" }}>
            <header style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '32px' }}>
                <h1 style={{ fontSize: '28px', fontWeight: 800, color: '#1e293b' }}>
                    {isExecutive ? "Global Pipeline" : `${user?.displayRole} Workspace`}
                </h1>
                <input 
                    type="text" placeholder="Search..." 
                    style={{ padding: '10px 16px', borderRadius: '12px', border: '1px solid #e2e8f0' }}
                    onChange={(e) => setSearchQuery(e.target.value)}
                />
            </header>
            {/* Premium Metrics Bar */}
            {(isExecutive || isApprovalRole) && (
                <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 1fr)', gap: '24px', marginBottom: '40px' }}>
                    <MetricCard label="Pipeline Total" val={metrics.total_value} icon="💰" color="#6366f1" isCurrency />
                    <MetricCard label="Pending Approval" val={counts?.PENDING_APPROVAL || 0} icon="⚡" color="#ef4444" />
                    <MetricCard label="Under Review" val={counts?.DRAFT || 0} icon="📝" color="#f59e0b" />
                    <MetricCard label="Completed" val={counts?.SUBMIT_WON || 0} icon="✅" color="#10b981" />
                </div>
            )}
            <div style={{ background: '#fff', borderRadius: '20px', boxShadow: '0 4px 20px rgba(0,0,0,0.05)', overflow: 'hidden' }}>
                <table style={{ width: '100%', borderCollapse: 'collapse' }}>
                    <thead style={{ background: '#f8fafc' }}>
                        <tr>
                            <th style={thStyle}>Opportunity</th>
                            <th style={thStyle}>Value</th>
                            <th style={thStyle}>Status</th>
                            <th style={thStyle}>Action</th>
                        </tr>
                    </thead>
                    <tbody>
                        {opportunities.map(opp => (
                            <tr key={opp.id} style={{ borderBottom: '1px solid #f1f5f9' }}>
                                <td style={tdStyle}>{opp.name} <br/><small>{opp.customer}</small></td>
                                <td style={tdStyle}>${opp.deal_value?.toLocaleString()}</td>
                                <td style={tdStyle}>{opp.workflow_status}</td>
                                <td style={tdStyle}><button onClick={() => navigate(`/opportunity/${opp.id}`)}>View</button></td>
                            </tr>
                        ))}
                    </tbody>
                </table>
            </div>
        </div>
    );
};

Task 4: Content Routing Consolidation
I have updated 

App.tsx
 to remove all fragmented legacy routes. All dashboard entries now redirect to the new /dashboard route.

tsx
function App() {
    return (
        <AuthProvider>
            <UserProvider>
                <Router>
                    <Layout>
                        <Routes>
                            <Route path="/" element={<HomeRedirect />} />
                            
                            {/* Unified Dashboard for all roles */}
                            <Route path="/dashboard" element={
                                <ProtectedRoute>
                                    <UnifiedDashboard />
                                </ProtectedRoute>
                            } />
                            {/* Legacy Aliases */}
                            <Route path="/management/*" element={<Navigate to="/dashboard" replace />} />
                            <Route path="/sales/*" element={<Navigate to="/dashboard" replace />} />
                            <Route path="/practice-head/*" element={<Navigate to="/dashboard" replace />} />
                            <Route path="/assigned-to-me" element={<Navigate to="/dashboard" replace />} />
                            <Route path="/sa/*" element={<Navigate to="/dashboard" replace />} />
                            
                            {/* Core Functional Routes */}
                            <Route path="/opportunity/:id" element={<ProtectedRoute><OpportunityDetail /></ProtectedRoute>} />
                            <Route path="/score/:id" element={<ProtectedRoute><ScoreOpportunity /></ProtectedRoute>} />
                        </Routes>
                    </Layout>
                </Router>
            </UserProvider>
        </AuthProvider>
    );
}
